			country_culture_group = jomon

				any_country = {
					num_of_cities >= 1
					tag = AIU
				}
			}

		}
		
		highlight = {

				}
			}

			owns_or_subject_owns_area = hokkaido_area

			custom_tooltip = "tier_2_formable_tooltip"

				limit = { is_tribal = yes }

				formable_capital_modifier_tribal_effect = yes

			base = 1
		}
	}
