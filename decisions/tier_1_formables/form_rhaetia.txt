country_decisions = {
	
	# Form Rhaetia
	form_rhaetia = {
		
		potential = {
			num_of_cities >= 1
			primary_culture = rhaetian
			NOT = {
				tag = RAE
			}
			OR = {
				is_ai = no
				num_of_cities >= 10
			}
		}
		
		highlight = {
			scope:province = {
				AND = {
					has_owner = yes
					owner = {
						primary_culture = rhaetian
					}
				}
			}
		}
		
		allow = {
			custom_tooltip = {
				text = formable_not_rhaetia_exists
				NOT = {
					any_country = {
						num_of_cities >= 1
						tag = RAE
					}
				}
			}
			custom_tooltip = {
				text = any_rhaetian_country_tt
				OR = {
					any_country = {
						count = all
						filter = {
							primary_culture = rhaetian
							num_of_cities >= 1
							NOT = {
								this = root
							}
						}
						is_subject_of = root
					}
					AND = {
						num_of_cities >= 1
						any_country = {
							count = 1
							primary_culture = rhaetian
						}
					}
				}
			}
			can_form_nation_trigger = yes
		}
		
		effect = {
			change_country_name = "RHAETIA_NAME"
			hidden_effect = {
				change_country_adjective = "RHAETIA_ADJECTIVE"
				change_country_color = rhaetia_color
				change_country_tag = RAE
			}
			capital_scope = {
				capital_formable_tribal_effect = yes
				formable_capital_modifier_tribal_effect = yes
			}
			if = {
				limit = {
					any_subject = {
						primary_culture = rhaetian
					}
				}
				every_subject = {
					limit = {
						primary_culture = rhaetian
					}
					every_owned_province = {
						limit = {
							OR = {
								is_in_region = rhaetia_region 
								is_in_region = cisalpine_gaul_region
							}
						}
						set_owned_by = root
					}
				}
			}
			add_2_free_province_investments = yes
		}
		
		ai_will_do = {
			base = 1
		}
	}

}
