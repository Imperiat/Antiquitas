namespace = games

games.1 = {
	type = country_event
	title = "games.1.t"
	desc = {
		random_valid = {
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = "games.1.desc_1"
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = "games.1.desc_2"
			}
		}
		first_valid = {
			triggered_desc = {
				trigger = {
					exists = scope:competitor
					has_variable = lunatic_competitor
				}
				desc = "games.1.desc_lunatic_ruler"
			}
			triggered_desc = {
				trigger = {
					exists = scope:competitor
					NOT = {
						has_variable = lunatic_competitor
					}
				}
				desc = "games.1.desc_normal_ruler"
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = "games.1.desc_not_competing"
			}
		}
		first_valid = {
			triggered_desc = {
				trigger = {
					has_local_variable = games_success
					local_var:games_success = 1 #Fail
					has_variable = lunatic_competitor
					exists = scope:competitor
				}
				desc = "games.1.desc_fail_lunatic"
			}
			triggered_desc = {
				trigger = {
					has_local_variable = games_success
					exists = scope:competitor
					local_var:games_success = 1 #Fail
				}
				desc = "games.1.desc_fail"
			}
			triggered_desc = {
				trigger = {
					has_local_variable = games_success
					exists = scope:competitor
					local_var:games_success = 2 #Success
				}
				desc = "games.1.desc_success"
			}
			triggered_desc = {
				trigger = {
					NOT = {
						exists = scope:competitor
					}
				}
				desc = "games.1.desc_not_competing_end"
			}
		}
	}
	
	picture = gladiators
	
	immediate = {
		random_owned_province = {
			weight = {
				modifier = {
					factor = 10
					is_state_capital = yes
				}
				modifier = {
					factor = 100
					controller = root
				}
			}
			save_scope_as = games_location
		}
		random_country = {
			limit = {
				NOT = {
					this = root
				}
			}
			save_scope_as = spooncrafter
		}
		set_local_variable = {
			name = games_success
			value = { integer_range = { min = 1 max = 2 } }
		}
	}
	
	option = {
		name = "games.1.a"
		trigger = {
			has_local_variable = games_success
			local_var:games_success = 1 #Fail
			has_variable = lunatic_competitor
			exists = scope:competitor
		}
		exclusive = yes
		current_ruler = {
			add_popularity = 5
			if = {
				limit = {
					health > 5
				}
				add_health = -5
			}
			else = {
				death = {
					death_reason = death_egg_and_spoon
				}
			}
		}
	}
	
	option = {
		name = "games.1.b"
		trigger = {
			has_local_variable = games_success
			exists = scope:competitor
			local_var:games_success = 1 #Fail
			NOT = {
				has_variable = lunatic_competitor 
			}
		}
		exclusive = yes
		current_ruler = {
			add_popularity = 5
			if = {
				limit = {
					health > 5
				}
				add_health = -5
			}
			else = {
				death = {
					death_reason = death_gladiatorial_combat
				}
			}
		}
	}
	
	option = {
		name = "games.1.c"
		trigger = {
			has_local_variable = games_success
			exists = scope:competitor
			local_var:games_success = 2 #Success
		}
		current_ruler = {
			add_popularity = 10
		}
	}
	
	option = { 
		name = "games.1.d"
		trigger = {
			NOR = {
				AND = {
					has_local_variable = games_success
					local_var:games_success = 1 #Fail
					has_variable = lunatic_competitor
					exists = scope:competitor
				}
				AND = {
					has_local_variable = games_success
					exists = scope:competitor 
					local_var:games_success = 1 #Fail
				}
				AND = {
					has_local_variable = games_success
					exists = scope:competitor
					local_var:games_success = 2 #Success
				}
			}
		}
		current_ruler = {
			add_popularity = 5
		}
	}   
	
	after = {
		if = {
			limit = {
				has_variable = lunatic_competitor
			}
			remove_variable = lunatic_competitor
		}
	}
	
}

games.2 = {
	type = country_event
	title = "games.2.t"
	desc = "games.2.desc"
	
	picture = hellenic_marketplace
	
	option = {
		name = "games.2.a"
		current_ruler = {
			save_scope_as = competitor
			add_popularity = 5
		}
	}
	
	option = {
		name = "games.2.b"
		trigger = {
			current_ruler = {
				age >= 60
			}
		}
		custom_tooltip = "old_ruler_not_good_at_sports"
		highlight = yes
	}
	
	option = {
		name = "games.2.c"
		trigger = {
			current_ruler = {
				age < 60
			}
		}
		current_ruler = {
			add_popularity = -5
		}
	}
	
	option = {
		name = "games.2.d"
		custom_tooltip = "lunatic_tooltip"
		trigger = {
			current_ruler = {
				has_trait = lunatic
			}
		}
		highlight = yes
		add_stability = -5
		current_ruler = {
			save_scope_as = competitor
			add_popularity = 10
		}
		set_variable = {
			name = lunatic_competitor
			days = 300
		}
	}
	
	after = {
		trigger_event = {
			id = games.1
			days = { 10 30 }
		}
	}
	
}
