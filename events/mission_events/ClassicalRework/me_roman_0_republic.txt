namespace = me_roman_0_republic

me_roman_0_republic.1 = {	# Deponse the Monarchy
    type = country_event
    title = me_roman_0_republic.1.t
    desc = me_roman_0_republic.1.desc
    picture = roman_trial
    left_portrait = current_ruler

    trigger = { has_land = yes }

    option = {      
        name = me_roman_0_republic.1.a

		custom_tooltip = me_roman_0_republic.1.a.tt

		hidden_effect = {
			c:ROM.current_ruler.family = {
				every_family_member = {
					set_as_minor_character = THIS
					add_triggered_character_modifier = {
						name = "was_once_great"
						duration = -1
					}
				}
				remove_family = ROOT
			}

			c:ROM = {
				change_government = aristocratic_republic
			}
			c:ROM.current_ruler = {
				c:ROM = {
					if = {
						limit = {
							any_neighbour_country = {
								always = yes
							}
						}
						random_neighbour_country = {
							save_scope_as = banishment_country
						}
					}
					else = {
						random_country = {
							save_scope_as = banishment_country
						}
					}
				}
				banish = scope:banishment_country
			}
		}
    }
}
