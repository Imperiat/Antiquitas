#The Return of Gelon
cr_sicily_flavour_events.2 = {
	type = country_event
	title = "cr_sicily_flavour_events.2.t"
	desc = "cr_sicily_flavour_events.2.desc"

	picture = great_work_supplies
	goto_location = p:94

	left_portrait = current_ruler

	trigger = {
		NOT = {
			has_global_variable = cr_treasure_charioteer_found
		}
		OR = {
			tag = SYR
			tag = MGG
		}
		is_monarchy = yes
		capital_scope = {
			this = p:84
		}
		owns = p:94
	}

	immediate = {

	}

	option = {	#Accept that statue
		name = "cr_sicily_flavour_events.2.a"
		set_global_variable = cr_treasure_charioteer_found
		add_treasury = {
			value = has_monthly_income
			multiply = -3.0
			max = 150
		}
		create_country_treasure = {
			key = cr_treasure_charioteer_of_motya
			icon = cr_treasure_charioteer_of_motya_01
			modifier = {
				local_happiness_for_same_culture_modifier = 0.03
			}
		}
		ai_chance = {
			factor = 30
			modifier = {
				treasury < 150
				factor = 0
			}
		}
	}

	option = {	#Reject the statue
		name = "cr_sicily_flavour_events.2.b"
		set_global_variable = cr_treasure_charioteer_found
		trigger = {
			current_ruler = {
				OR = {
					NOR = {
						has_trait = scholar
						has_trait = sceptical
						finesse >=10
					}
					has_trait = blind
					has_trait = brain_damaged
					has_trait = dementia
				}
			}
		}

		current_ruler = {
			add_popularity = subtract_popularity_small
		}

		ai_chance = {
			factor = 10
			modifier = {
				treasury < 150
				factor = 2
			}
		}
	}

	option = {	#sceptical ruler
		name = "cr_sicily_flavour_events.2.c"
		set_global_variable = cr_treasure_charioteer_found
		highlight = yes
		trigger = {
			current_ruler = {
				OR = {
					has_trait = scholar
					has_trait = sceptical
					finesse >=10
				}

				NOT = {
					has_trait = blind
					has_trait = brain_damaged
					has_trait = dementia
				}
			}
		}
		if = {
			limit = {
				current_ruler = {
					has_trait = scholar
				}
			}
			custom_tooltip = "cr_sicily_flavour_events.2b_tt_a"
		}
		if = {
			limit = {
				current_ruler = {
					NOT = { has_trait = scholar }
					has_trait = sceptical
				}
			}
			custom_tooltip = "cr_sicily_flavour_events.2b_tt_b"
		}
		if = {
			limit = {
				current_ruler = {
					NOR = {
						has_trait = scholar
						has_trait = sceptical
					}
					finesse >=10
				}
			}
			custom_tooltip = "cr_sicily_flavour_events.2b_tt_c"
		}
		if = {
			limit = {
				current_ruler = {
					has_trait = trusting
				}
			}
			current_ruler = {
				remove_trait = trusting
			}
		}
		else_if = {
			limit = {
				current_ruler = {
					has_trait = guileless
				}
			}
			current_ruler = {
				remove_trait = guileless
			}
		}
		else_if = {
			limit = {
				NOT = {
					current_ruler = {
						has_trait = sceptical
						has_trait = pious
					}
				}
			}
			current_ruler = {
				add_trait = sceptical
			}
		}
		else_if = {
			limit = {
				always = yes
			}
			current_ruler = {
				add_finesse = 1
			}
		}
		ai_chance = {
			factor = 50
		}
	}

	option = {	#Accept that statue
		name = "cr_sicily_flavour_events.2.d"
		set_global_variable = cr_treasure_charioteer_found
		custom_tooltip = cr_sicily_flavour_events.2.d_tt
		highlight = yes
		trigger = {
			current_ruler = {
				OR = {
					has_trait = arbitrary
					has_trait = corrupt
					has_trait = cruel
				}
				NOT = {
					has_trait = benefactor
					has_trait = generous
					has_trait = good_natured
					has_trait = honest
					has_trait = just
					has_trait = kindly
					has_trait = righteous
				}
			}
		}
		add_tyranny = tyranny_small
		create_country_treasure = {
			key = cr_treasure_charioteer_of_motya
			icon = cr_treasure_charioteer_of_motya_01
			modifier = {
			local_happiness_for_same_culture_modifier = 0.03
			}
		}
		if = {
			limit = {
				current_ruler = {
					has_trait = just
				}
			}
			current_ruler = {
				remove_trait = just
			}
		}
		ai_chance = {
			factor = 25
		}
	}

}

						current_ruler.family = fam:Magid

					}
				}
				desc = cr_sicily_flavour_events.11.desc_magid
			}

				trigger = {
					always = yes
				}

			}
		}

	goto_location = p:94

		exists = c:CAR
	}

			}
		}

			limit = {
				current_co_ruler.family = fam:Magid
			}

			}

			modifier = {

			}

		}
	}


	option = {	#Take the statue for yourself
		name = "cr_sicily_flavour_events.11.b"
		add_treasury = {
			value = has_monthly_income
			multiply = -3.0
			max = 150
		}
		create_country_treasure = {
			key = cr_treasure_charioteer_of_motya

			}
		}

		ai_chance = {
			factor = 30
			modifier = {
				treasury < 150
				factor = 0
			}
		}
	}

}
