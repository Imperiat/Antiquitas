death.2 = {
	type = character_event
	title = "death.2.t"
	desc = "death.2.desc"
	picture = greek_court
	
	left_portrait = root
	
	trigger = {
		is_alive = yes
		age > 65
		NOR = {
			has_trait = dementia
			has_trait = cancer
			has_trait = gout
			has_trait = diphtheria
		}
	}
	
	weight_multiplier = {
		modifier = {
			factor = 3
			age > 70
		}
		modifier = {
			factor = 3
			age > 75
		}
	}
	
	immediate = {
		hidden_effect = {
			random_list = {
				25 = {
					trigger = {
						NOT = {
							has_trait = dementia
						}
					}
					set_local_variable = {
						name = terminal_disease_temp
						value = 1
					}
				}
				25 = {
					trigger = {
						NOT = {
							has_trait = cancer
						}
					}
					set_local_variable = {
						name = terminal_disease_temp
						value = 2
					}
				}
				25 = {
					trigger = {
						NOT = {
							has_trait = gout
						}
					}
					set_local_variable = {
						name = terminal_disease_temp
						value = 3
					}
				}
				25 = {
					trigger = {
						NOT = {
							has_trait = diphtheria
						}
					}
					set_local_variable = {
						name = terminal_disease_temp
						value = 4
					}
				}
			}
		}
	}
	
	option = {
		name = "death.2.a"
		if = {
			limit = {
				local_var:terminal_disease_temp = 1
			}
			add_trait = dementia
		}
		else_if = {
			limit = {
				local_var:terminal_disease_temp = 2
			}
			add_trait = cancer
		}
		else_if = {
			limit = {
				local_var:terminal_disease_temp = 3
			}
			add_trait = gout
		}
		else_if = {
			limit = {
				local_var:terminal_disease_temp = 4
			}
			add_trait = diphtheria
		}
	}
}
