#Imprisoned Rival does bad things(tm)
ip_generic.1 = {
	type = character_event
	hidden = yes
	
	trigger = {
		exists = employer.current_ruler
		prisoner = yes
		can_get_rivals = yes
		is_adult = yes
		NOT = {
			has_variable = ip_generic_1_toggle
		}
		employer = {
			NOT = {
				has_variable = ip_generic_1_toggle
			}
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			factor = 2
			has_trait = assertive
		}
		modifier = {
			factor = 2
			has_trait = crafty
		}
	}
	
	immediate = { 
		save_scope_as = ip_generic_1_antagonist
		set_variable = {
			name = ip_generic_1_toggle
			days = -1
		}
		if = {
			limit = {
				NOT = {
					is_rival = employer.current_ruler
				}
				employer.current_ruler = {
					can_get_rivals = yes
				}
				can_get_rivals = yes
			}
			add_rival = employer.current_ruler
		}
		employer = {
			current_ruler = {
				trigger_event = {
					id = ip_generic.2
				}
			}
		}
	}
}

	desc = "ip_generic.21.desc"
	picture = farming
	
	left_portrait = root.current_ruler
	right_portrait = scope:soldiers_representative
	
	trigger = {
		NOT = {
			has_variable = ip_21_toggle
		}
		OR = {
			AND = {
				has_war_exhaustion > 6
				war = no
			}
			stability <= 60
		}
		any_character = {
			trigger_if = {
				limit = {
					root = {
						is_monarchy = yes
					}
				}
				has_office = office_master_of_the_guard
			}
			trigger_if = {
				limit = {
					root = {
						is_republic = yes
					}
				}
				has_office = office_tribune_of_the_soldiers
			}
			trigger_if = {
				limit = {
					root = {
						is_tribal = yes
					}
				}
				has_office = office_warchief
			}
		}
		any_character = {
			num_loyal_cohorts >= 2
			is_mercenary = no
		}
	}
	
	immediate = {
		set_variable = {
			name = ip_21_toggle
			days = 14600
		}
		random_character = {
			limit = {
				trigger_if = {
					limit = {
						root = {
							is_monarchy = yes
						}
					}
					has_office = office_master_of_the_guard
				}
				trigger_if = {
					limit = {
						root = {
							is_republic = yes
						}
					}
					has_office = office_tribune_of_the_soldiers
				}
				trigger_if = {
					limit = {
						root = {
							is_tribal = yes
						}
					}
					has_office = office_warchief
				}
			}
			save_scope_as = soldiers_representative
		}
		set_variable = {
			name = num_of_loyal_cohorts_total
			value = 0
		}
		while = {
			limit = {
				any_character = {
					is_general = yes
					is_mercenary = no
					num_loyal_cohorts >= 1
					NOT = {
						has_variable = ip_21_counted
					}
				}
			}
			random_character = {
				limit = {
					is_general = yes
					is_mercenary = no
					num_loyal_cohorts >= 1
					NOT = {
						has_variable = ip_21_counted
					}
				}
				weight = {
					title_importance_modifier = yes
				}
				save_scope_as = countee
				set_variable = {
					name = ip_21_counted
					days = 1
				}
			}
			change_variable = {
				name = num_of_loyal_cohorts_total
				add = scope:countee.num_loyal_cohorts
			}
		}
	}
	
	option = {
		name = "ip_generic.21.a"
		trigger = {
			any_owned_province = {
				count >= half_all_loyal_cohorts_svalue
				has_province_modifier = state_owned_farmlands
			}
		}
		ordered_owned_province = {
			limit = {
				has_province_modifier = state_owned_farmlands
			}
			order_by = total_population
			max = {
				value = all_loyal_cohorts_svalue
				divide = 2
				subtract = 1
			}
			remove_province_modifier = state_owned_farmlands
		}
	}
	
	option = {
		name = "ip_generic.21.b" 
		trigger = {
			any_owned_province = {
				has_province_modifier = state_owned_farmlands
			}
			any_owned_province = {
				count < half_all_loyal_cohorts_svalue
				has_province_modifier = state_owned_farmlands
			}
		}
		custom_tooltip = "ip_21_number_tooltip_2" 
		every_owned_province = {
			limit = {
				has_province_modifier = state_owned_farmlands
			}
			remove_province_modifier = state_owned_farmlands
		}
	}
	
	option = {
		name = "ip_generic.21.c"
		every_character = {
			limit = {
				is_general = yes
				is_mercenary = no
			}
			if = {
				limit = {
					is_ruler = no
				}
				add_loyalty = loyalty_bribed_commander
			}
			ordered_character_unit = {
				order_by = unit_size
				add_unit_modifier = {
					name = morale_malus
					duration = 1440
				}
			}
		}
		add_treasury = {
			value = 0
			subtract = root.has_monthly_income
			multiply = 4
		}
	}
	
	option = {
		name = "ip_generic.21.d"
		every_character = {
			limit = {
				is_general = yes
				is_mercenary = no
			}
			if = {
				limit = {
					is_ruler = no
				}
				add_loyalty = loyalty_veterans_neglected
			}
			ordered_character_unit = {
				order_by = unit_size
				add_unit_modifier = {
					name = morale_malus
					duration = 1440
				}
			}
		}
		remove_stability_effect = yes
	}
	
	option = {
		name = "ip_generic.21.e"
		highlight = yes
		custom_tooltip = lunatic_tooltip
		trigger = {
			current_ruler = {
				has_trait = lunatic
			}
		}
		current_ruler = {
			if = {
				limit = {
					number_of_status_traits < max_number_of_status_traits_svalue
					NOT = {
						has_trait = victorious
					}
				}
				add_trait = victorious
			}
			else = {
				add_popularity = popularity_huge
			}
		}
		every_character = {
			limit = {
				is_general = yes
				is_ruler = no
				is_mercenary = no
			}
			add_loyalty = loyalty_god_incarnate
			ordered_character_unit = {
				order_by = unit_size
				add_unit_modifier = {
					name = morale_malus
					duration = 1440
				}
			}
		}
		remove_stability_effect = yes
	}
}
