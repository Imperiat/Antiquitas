namespace = techspread

techspread.1 = { #
	type = country_event
	hidden = yes

	trigger = {
		is_tribal = yes
		oratory_tech < 5
	}
	immediate = {
		random_country = {
			limit = {
				is_tribal = no
				OR = {
					exports_to = root
					imports_from = root
				}
			}
			save_scope_as = techtarget
		}
		if = {
			limit = {
				exists = scope:techtarget	
			}
			trigger_event =  {
				id = techspread.2
			}
		}
	}

}
