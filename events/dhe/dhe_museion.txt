namespace = dhe_museion

# Museion maintenance event
dhe_museion.1 = {
	type = country_event
	hidden = yes

	trigger = {
		owns = 516
	}

	immediate = {
		if = {
			limit = { # Check Progress Tracker
				owns = 516
				var:museion_progress < 3
			}
			random_list = {
				1 = { # Progress - No Event
				}
				10 = { # Progress Event - Academics
					trigger = {
						any_country = {
							count > 2
							NOT = {
								THIS = root
							}
							any_character = {
								is_ruler = no
								is_pretender = no
								age >= 15
								has_trait = scholar
								char_is_hellenic_sphere_trigger = yes
							}
						}
					}
					trigger_event = {
						id = dhe_museion.2
					}
				}
				10 = { # Progress Event - Books
					trigger = {
						NOT = {
							has_country_modifier = country_improvement_enormous_library
						}
					}
					trigger_event = {
						id = dhe_museion.3
					}
				}
				10 = { # Progress Event - Poets
					trigger = {
						any_country = {
							count > 1
							NOT = {
								THIS = root
							}
							any_character = {
								is_ruler = no
								is_pretender = no
								age >= 15
								has_trait = poet
								char_is_hellenic_sphere_trigger = yes
							}
						}
					}
				}
				5 = { # Progress Event - Obscurities
					trigger = {
						NOT = {
							has_country_modifier = country_improvement_rare_text
						}
					}
					trigger_event = {
						id = dhe_museion.5
					}
				}
			}
			change_variable = { # Bump Progress Tracker
				name = museion_progress
				add = 1
			}
			trigger_event = { # Launch the event again
				id = dhe_museion.1
				days = { 200 400 }
			}
		}
		else_if = { # Building Complete
			limit = {
				owns = 516
				var:museion_progress >= 3
			}
			trigger_event = {
				id = dhe_museion.11
			}
		}
	}
}
