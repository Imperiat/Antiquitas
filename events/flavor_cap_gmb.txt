namespace = flavor_cap

# Cappadocia Starting Event
flavor_cap.1 = {
	type = country_event
	title = "flavor_cap.1.t"
	desc = "flavor_cap.1.desc"
	picture = romans_marching

	fire_only_once = yes


	left_portrait = root.current_ruler
	right_portrait = char:433
	right_portrait = char:50


	trigger = {
		tag = CAP
	}


	immediate = {
		hidden_effect = {
			current_ruler = {
				save_scope_as = old_ruler
			}
		}
	}
	# Bring back Eumenes
	option = {
		name = flavor_cap.1.a
		ai_chance = { factor = 10 }
		if = {
            limit = {
                NOT = {
                    primary_culture = greco_pontic
                }
            }
            set_primary_culture_cleanup_effect = {
                NEW_PRIMARY_CULTURE = greco_pontic
                MAKE_OLD_CULTURE_INTEGRATED = yes
            }
        }
		hidden_effect = {
			current_ruler = {
				remove_all_positions = yes
				#set_variable = eumenes_mission_tree_1_unlock
			}
			c:PRY = {
				trigger_event = {
					id = flavor_cap.2
				}
			}
		}
		trigger = {
			char:433 = {
				is_alive = yes
				is_ruler = no
				employer = c:CAP
			}
		}
		set_as_ruler = char:433
		scope:old_ruler = {
			death = {
				killer = char:433
				death_reason = death_murder
			}
		}
	}
	# Bring back the Persians
	option = {
		name = flavor_cap.1.b
		ai_chance = { factor = 10 }
		set_as_ruler = char:50
		if = {
            limit = {
                NOT = {
                    primary_culture = cappadocian
                }
            }
            set_primary_culture_cleanup_effect = {
                NEW_PRIMARY_CULTURE = cappadocian
                MAKE_OLD_CULTURE_INTEGRATED = yes
            }
        }
		set_country_religion = zoroaster
		char:50 = {
			hidden_effect = {
				remove_all_positions = yes
				move_country = c:CAP
				set_as_minor_character = THIS
				remove_character_modifier = foreign_citizen
				set_home_country = c:CAP
				c:PRY = {
					trigger_event = { id = flavor_cap.5 }
				}
			}
		}
		char:433 = {
			death = {
				killer = CHAR:51
				death_reason = death_murder
			}
		}
		scope:old_ruler = {
			death = {
				killer = char:50
				death_reason = death_murder
			}
		}
		hidden_effect = {
			every_character = {
				limit = {
					has_culture = cappadocian
				}
				add_loyalty = loyalty_flavor_arm_ariarathes_reinstated
			}
		}
	}
	# Stay true to the Antigonids
	option = {
		name = flavor_cap.1.c
		ai_chance = { factor = 80 }
		char:433 = {
			death = {
				killer = CHAR:50
				death_reason = death_murder
			}
		}
		char:50 = {
			death = {
				killer = scope:old_ruler
				death_reason = death_murder
			}
		}
		hidden_effect = {
			every_character = {
				limit = {
					has_culture = cappadocian
				}
				add_loyalty = loyalty_flavor_arm_ariarathes_killed

			}
			char:50 = {
				remove_all_positions = yes
			}
			c:PRY = {
				trigger_event = { id = flavor_cap.4 }
			}
		}
	}

}
# Antigonid Event, Eumenid Coup in Cappadocia
flavor_cap.2 = {
	type = country_event
	title = "flavor_cap.2.t"
	desc = "flavor_cap.2.desc"
	picture = army_desert
	fire_only_once = yes
	left_portrait = root.current_ruler
	right_portrait = char:433
	immediate = {
		hidden_effect = {
			c:CAP = { capital_scope = { save_scope_as = cap_capital } }
		}
   	}

	# War against CAP is not declared
	option = {
		ai_chance = { factor = 90 }
		name = flavor_cap.2.a
		trigger = { war = no }
		release_subject = c:CAP
		add_truce = {
			target = c:CAP
			duration = 1825
		}
		hidden_effect = {
			c:CAP = {
				add_truce = {
					target = ROOT
					duration = 1825
				}
				every_owned_province = {
					add_claim = ROOT
				}
			}
		}

	}
	# War against CAP is declared
	option = {
		name = flavor_cap.2.b
		ai_chance = { factor = 10 }
		trigger = { war = no }
		release_subject = c:CAP
        declare_war_with_wargoal = {
			war_goal = conquer_wargoal
			province = scope:cap_capital
			target = c:CAP
		}
		trigger_event = {
			id = flavor_cap.3
		}
	}
}
# War is declared
flavor_cap.3 = {
	type = country_event
	hidden = yes

	immediate = {
		random_current_war = {
			limit = {
				any_war_defender = {
					tag = CAP
				}
			}
			save_scope_as = cappadocian_war_scope
		}
		every_subject = {
			limit = {
				OR = {
					is_subject_type = feudatory
					is_subject_type = satrapy
					is_subject_type = client_state
				}
				has_subject_loyalty >= 33
				save_temporary_scope_as = temp_subject_war_scope
				scope:cappadocian_war_scope = {
					NOR = {
						any_war_attacker = { this = scope:temp_subject_war_scope }
						any_war_defender = { this = scope:temp_subject_war_scope }
					}
				}
			}
			add_to_war = {
				target = scope:cappadocian_war_scope
				attacker = yes
				leader = no
			}
		}
	}
}

		}

			c:CAP = {
				add_truce = {
					target = ROOT
					duration = 1825
				}
				every_owned_province = {
					add_claim = ROOT
				}
			}
		}
    }
	# attack Cappadocia
    option = {
        name = flavor_cap.5.b
        ai_chance = { factor = 10 }
		trigger = { war = no }
		release_subject = c:CAP
        declare_war_with_wargoal = {
			war_goal = conquer_wargoal
			province = scope:cap_capital
			target = c:CAP
		}
		trigger_event = {
			id = flavor_cap.3

}
