johan.2 = {

		owner = { tag = ROM }
		is_capital = yes
		is_port = no
		
		owner = {
			num_of_unit_type = {
				type = light_infantry
				value < 5
			}
			
			num_of_unit_type = {
				type = heavy_infantry
				value >= 2
			}
		}
	}	
	
	
	title = EPI_Events.1.t
	desc = EPI_Events.1.d
	


	option = {
		name = EPI_Events.1.a
		random_pops_in_province = {
			limit = {
				pop_type = citizen
			}
			set_pop_religion = druidism
			set_pop_culture = macedonian
		}
		add_building_level = fortress_building
		create_pop = slaves
	}
	
	option = {
		name = "move_the_pops"
		p:2 = {
			every_pops_in_province = {
				limit = {
					pop_type = citizen
				}
				move_pop = 1
			}
		}
	}
}
