deficit_civic_neglect_button = {
	
	scope = country
	
	is_shown = { # Can be omitted as always true
      always = yes
    } 
	ai_is_valid = {
		always = no #TODO PWN
	}
	is_valid = {
		any_owned_province = {
			has_city_status = yes
			has_building = housing_building
		# always = yes
		}
		NOT = {
			has_country_modifier = aqueducts_in_disrepair_manual
		}
	}
	effect = {
		# custom_tooltip = "deficit_civic_neglect_button_tt"
		add_treasury  = {
			value = root.has_monthly_income
			multiply = add_monthly_income_small
			min = 20
			max = 1200
		}
		add_stability = -5 
		hidden_effect = {
			remove_country_modifier = aqueducts_in_disrepair_manual
		}
		add_country_modifier = {
			name = "aqueducts_in_disrepair_manual"
			duration = 1825
		}
		random_owned_province = {
			limit = {
				has_city_status = yes
				has_building = housing_building
			}
			while = {
				count = 3 # has_building = housing_building 
				remove_building_level = housing_building 
			}
			random_pops_in_province = {
				kill_pop = yes
			}
			random_pops_in_province = {
				kill_pop = yes
			}
			add_province_modifier = {
				name = being_extorted
				duration = 1095
			}
		}
	}
}
