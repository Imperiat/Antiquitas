rm_rome_subjects_allies_cohorts_svalue = {
	value = 0
	if = {
		limit = {
			c:ROM = { 
				has_land = yes
				num_of_cohorts > 0 
			}
		}
		c:ROM = {
			every_unit = { 
				every_sub_unit = { add = 1 }
			}
		}
	}
	if = {
		limit = {
			any_country = { is_subject_of = c:ROM }
		}
		c:ROM = {
			every_subject = {
				limit = {
					OR = {
						is_subject_type = feudatory
						is_subject_type = client_state
					}
					num_of_cohorts > 0 
				}
				every_unit = { 
					every_sub_unit = { add = 1 }
				}
			}
		}
	}
	if = {
		limit = {
			any_country = { alliance_with = c:ROM }
		}
		c:ROM = {
			every_allied_country = {
				limit = {
					has_land = yes
					num_of_cohorts > 0 
				}
				every_unit = { 
					every_sub_unit = { add = 1 }
				}
			}
		}
	}
	divide = 2
}
