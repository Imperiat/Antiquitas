# Get a random event picture for a culture group
# Use this effect in the immediate to use the pictures in 000_mil_pictures.txt
# cg = culture group to use, can be:

# britannic
# gallic
# germanic
# hellenic
# latin
# south_levantine
# indian
# persia
# scythia
# north_african (this is actually egyptian)
# numidian/fezzani (this is actually north african)

get_event_picture = {
	if = {
		limit = {
			culture_group:$cg$ = culture_group:britannic
		}
		set_variable = {
			name = event_picture
			value = {
				integer_range = {
					min = 1
					max = 8
				}
			}
			days = 1
		}
	}
	if = {
		limit = {
			culture_group:$cg$ = culture_group:gallic
		}
		set_variable = {
			name = event_picture
			value = {
				integer_range = {
					min = 1
					max = 8
				}
			}
			days = 1
		}
	}
	if = {
		limit = {
			culture_group:$cg$ = culture_group:germanic
		}
		set_variable = {
			name = event_picture
			value = {
				integer_range = {
					min = 1
					max = 9
				}
			}
			days = 1
		}
	}
	if = {
		limit = {
			culture_group:$cg$ = culture_group:hellenic
		}
		set_variable = {
			name = event_picture
			value = {
				integer_range = {
					min = 1
					max = 26
				}
			}
			days = 1
		}
	}
	if = {
		limit = {
			culture_group:$cg$ = culture_group:latin
		}
		set_variable = {
			name = event_picture
			value = {
				integer_range = {
					min = 1
					max = 25
				}
			}
			days = 1
		}
	}
	if = {
		limit = {
			culture_group:$cg$ = culture_group:south_levantine
		}
		set_variable = {
			name = event_picture
			value = {
				integer_range = {
					min = 1
					max = 9
				}
			}
			days = 1
		}
	}
	if = {
		limit = {
			culture_group:$cg$ = culture_group:north_african
		}
		set_variable = {
			name = event_picture
			value = {
				integer_range = {
					min = 1
					max = 22
				}
			}
			days = 1
		}
	}
	if = {
		limit = {
			culture_group:$cg$ = culture_group:indian
		}
		set_variable = {
			name = event_picture
			value = {
				integer_range = {
					min = 1
					max = 25
				}
			}
			days = 1
		}
	}
	if = {
		limit = {
			culture_group:$cg$ = culture_group:persia
		}
		set_variable = {
			name = event_picture
			value = {
				integer_range = {
					min = 1
					max = 16
				}
			}
			days = 1
		}
	}
	if = {
		limit = {
			culture_group:$cg$ = culture_group:scythia
		}
		set_variable = {
			name = event_picture
			value = {
				integer_range = {
					min = 1
					max = 11
				}
			}
			days = 1
		}
	}
	if = {
		limit = {
			OR = {
				culture_group:$cg$ = culture_group:fezzani
				culture_group:$cg$ = culture_group:numidian
			}
		}
		set_variable = {
			name = event_picture
			value = {
				integer_range = {
					min = 1
					max = 25
				}
			}
			days = 1
		}
	}
	else = {
		set_variable = {
			name = event_picture
			value = {
				integer_range = {
					min = 1
					max = 8
				}
			}
			days = 1
		}
	}
}
