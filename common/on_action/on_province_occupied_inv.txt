on_province_occupied = { #province scope, it will fire AS WELL as on_siege_won
	effect = {
		###Invictus Added - Barbarians get Inventions
		if = {
			limit = {
				NOT = {
					has_province_modifier = disturbed_circles
				}
				OR = {
					has_province_rank = city_metropolis
					has_city_status = yes
				}
				trigger_if = {
					limit = {
						current_date >= 354.1.1
					}
					OR = {
						total_population >= 50
						has_province_rank = city_metropolis
						is_capital = yes
					}
					civilization_value >= 40
				}
				trigger_else_if = {
					limit = {
						current_date >= 254.1.1
						current_date < 354.1.1
					}
					OR = {
						total_population >= 40
						has_province_rank = city_metropolis
						is_capital = yes
					}
					civilization_value >= 30
				}
				trigger_else_if = {
					limit = {
						current_date >= 154.6.1
						current_date < 254.1.1
					}
					OR = {
						total_population >= 30
						has_province_rank = city_metropolis
						is_capital = yes
					}
					civilization_value >= 20
				}
				controller = {
					is_tribal = yes
				}
			}
			random = {
				chance = 20
				controller = {
					trigger_event = {
						id = tribal_fun_inv.1
						days = { 10 80 }
					}
				}
				add_province_modifier = {
					name = disturbed_circles
					duration = 3650 #10yrs
				}
			}
		}
		#easter egg when sacking Syracuse; added by Invictus
		if = {
			limit = {
				this = p:84
				current_date > 500.1.1
				NOT = {
					exists = treasure:do_not_disturb_these_circles
				}
			}
			controller = {
				create_country_treasure = {
					key = do_not_disturb_these_circles
					icon = treasure_chest
					modifier = {
						local_research_points_modifier = 0.25
					}
				}
			}
		}
	}

	events = {
	}
}
