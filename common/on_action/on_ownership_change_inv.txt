on_ownership_change = { #province scope
	effect = {
		if = {
			limit = {
				exists = owner
				owner = {
					country_culture_group = chinese
				}
			}
			owner = {
				check_chinese_unification_provs_effect = yes
				check_chinese_unification_subjects_effect = yes
			}
		}
		if = {
			limit = {
				is_iron_man = yes
				exists = owner
				owner = {
					is_ai = no
					has_land = yes
				}
			}
			owner = {
				if = {
					limit = {
						NOT = {
							has_variable = ach_render_unto_caesar_var
						}
					}
					set_variable = {
						name = ach_render_unto_caesar_var
					}
				}
			}
		}
		if = {
			limit = {
				exists = owner
				owner = {
					has_land = yes
					tag = MAC
				}
				has_variable = mac_city_rename_var
			}
			trigger_event = {
				id = mac_rename_events.1
			}
		}
		if = {
			limit = {
				exists = holding_owner
			}
			holding_owner = {
				remove_holding = root
			}
		}
	}
}
