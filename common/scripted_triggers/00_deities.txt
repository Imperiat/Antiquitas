### General Deity Checks ###
very_common_deity_trigger = {
	OR = {
		religion = $RELIGION$
		religion_pops_in_country = {
			target = $RELIGION$
			value >= very_common_deity_svalue
		}
	}
}

can_deify_trigger = {
	root.current_ruler = {
		popularity >= 90
		custom_tooltip = {
			text = prestige_deification_requirement
			family = { prestige >= 800 }
		}
	}
}
